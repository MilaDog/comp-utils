257        """Traverse the grid to find the first instance of `value`, returning the position of the element.

286        """Get an iterable of all the positions in the grid.

295    def get_neighbours(self, row: int, col: int, include_diagonals: bool = False) -> Iterable[tuple[int, int]]:
296        """Get the positions all of neighbouring cells.
297
298        Args:
299            row (int): Row index.
300            col (int): Column index.
301            include_diagonals (bool): Include diagonal neighbours.
302
303        Returns:
304            Iterable[tuple[int, int]]: All neighbouring cell positions.
305        """
306        directions: list[tuple[int, int]] = [(-1, 0), (1, 0), (0, -1), (0, 1)]
307        if include_diagonals:
308            directions += [(-1, -1), (-1, 1), (1, -1), (1, 1)]
309
310        for dx, dy in directions:
311            new_x, new_y = row + dx, col + dy
312
313            if self._is_valid(new_x, new_y):
314                yield new_x, new_y

328        """Filter grid cells by predicate, replacing non-matching with empty string.
